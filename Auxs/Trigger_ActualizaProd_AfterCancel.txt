CREATE TRIGGER `trg_ProductosVentaCancel` AFTER UPDATE ON `ventas` FOR EACH ROW BEGIN

UPDATE PRODUCTOS T1 SET T1.PROD_CANT = T1.PROD_CANT + (SELECT VENDET_CANTIDAD FROM VENTA_DETALLE WHERE VENDET_FOLIO = NEW.VENT_IDFOLIO AND VENDET_PRODUCTOID = T1.PROD_ID )
WHERE T1.PROD_ID =  (SELECT VENDET_PRODUCTOID FROM VENTA_DETALLE WHERE VENDET_FOLIO = NEW.VENT_IDFOLIO AND VENDET_PRODUCTOID = T1.PROD_ID);

END;;
